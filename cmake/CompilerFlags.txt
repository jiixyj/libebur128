foreach(_warning ${FLAGS})
  string(REGEX REPLACE "[^a-zA-Z0-9]" "" __warning ${_warning})
  string(TOUPPER ${__warning} __warning)
  CHECK_C_COMPILER_FLAG(${_warning} ${__warning})
  if(${__warning})
    set(VALID_FLAGS "${VALID_FLAGS} ${_warning}")
  endif()
endforeach()
